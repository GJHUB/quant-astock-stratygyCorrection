import pandas as pd
import numpy as np
import json
from itertools import product
import math

# ==========================================
# 补丁 1：JSON Numpy 类型转换器
# 解决 TypeError: Object of type int64 is not JSON serializable
# ==========================================
def convert_np_types(obj):
    if isinstance(obj, np.integer):
        return int(obj)
    elif isinstance(obj, np.floating):
        return float(obj)
    elif isinstance(obj, np.ndarray):
        return obj.tolist()
    elif isinstance(obj, dict):
        return {k: convert_np_types(v) for k, v in obj.items()}
    elif isinstance(obj, list):
        return [convert_np_types(i) for i in obj]
    return obj

# ==========================================
# 补丁 2：全局指标预计算 (彻底解决测试集 0 交易)
# ==========================================
def calculate_global_indicators(df):
    """
    重要改动：在进行 WFO 日期切割之前，先在整张全量数据表上计算所有指标。
    这样切分出来的测试集第一天，就已经自带了前 60 天计算好的均线和 ATR，无需预热！
    """
    df = df.sort_values(['tic', 'date']).copy() # 确保按股票和时间排序
    
    # 计算均线与乖离率 (Bias)
    df['SMA_20'] = df.groupby('tic')['close'].transform(lambda x: x.rolling(window=20).mean())
    df['SMA_60'] = df.groupby('tic')['close'].transform(lambda x: x.rolling(window=60).mean())
    df['SMA_60_shift5'] = df.groupby('tic')['SMA_60'].transform(lambda x: x.shift(5))
    df['BIAS_20'] = (df['close'] - df['SMA_20']) / df['SMA_20'] * 100
    df['Vol_SMA_10'] = df.groupby('tic')['volume'].transform(lambda x: x.rolling(window=10).mean())
    
    # 计算 ATR (Average True Range)
    df['prev_close'] = df.groupby('tic')['close'].transform(lambda x: x.shift(1))
    df['TR'] = np.maximum(
        df['high'] - df['low'],
        np.maximum(
            abs(df['high'] - df['prev_close']),
            abs(df['low'] - df['prev_close'])
        )
    )
    df['ATR_14'] = df.groupby('tic')['TR'].transform(lambda x: x.rolling(window=14).mean())
    
    # 计算长上影线比例
    df['Upper_Shadow'] = df['high'] - df[['close', 'open']].max(axis=1)
    df['Body_Total'] = df['high'] - df['low'] + 1e-8
    df['Upper_Shadow_Ratio'] = df['Upper_Shadow'] / df['Body_Total']
    
    return df

# ==========================================
# 信号发生器与单次回测评估
# ==========================================
def evaluate_strategy(df_slice, theta_buy, theta_sell, alpha_vol):
    """
    在已切割好的时间窗口内，根据传入的参数生成信号并计算表现
    """
    df = df_slice.copy()
    
    # 向量化生成买卖信号
    buy_condition = (
        (df['SMA_60'] > df['SMA_60_shift5']) & 
        (df['BIAS_20'] < -theta_buy) & 
        (df['volume'] < alpha_vol * df['Vol_SMA_10']) &
        (df['close'] > df['open']) 
    )
    
    sell_condition = (
        (df['BIAS_20'] > theta_sell) | 
        ((df['volume'] > 2.0 * df['Vol_SMA_10']) & (df['Upper_Shadow_Ratio'] > 0.6)) |
        (df['close'] < df['SMA_60'] * 0.95) 
    )
    
    df['Signal'] = 0
    df.loc[buy_condition, 'Signal'] = 1
    df.loc[sell_condition, 'Signal'] = -1
    
    # --- 简化的伪回测统计 (供快速参数寻优使用) ---
    # 统计买入触发次数作为总交易笔数
    total_trades = int(df[df['Signal'] == 1].shape[0])
    
    # 假定持有 5 天的极简收益率计算（实际应使用复杂的仓位引擎，这里为加快寻优速度做简化）
    df['Return_5d'] = df.groupby('tic')['close'].transform(lambda x: x.shift(-5) / x - 1)
    
    if total_trades > 0:
        win_trades = int((df[(df['Signal'] == 1) & (df['Return_5d'] > 0)]).shape[0])
        win_rate = win_trades / total_trades
        avg_return = df[df['Signal'] == 1]['Return_5d'].mean()
        # 近似最大回撤 (简化版)
        max_dd = abs(df[df['Signal'] == 1]['Return_5d'].min()) if total_trades > 0 else 1.0
        
        # ==========================================
        # 补丁 3：交易笔数惩罚机制
        # ==========================================
        if total_trades < 10:
            calmar_ratio = 0.0  # 惩罚：交易次数太少，不具统计学意义
        else:
            calmar_ratio = avg_return / (max_dd + 1e-6)
    else:
        win_rate, avg_return, calmar_ratio = 0.0, 0.0, 0.0
        
    return {
        'total_trades': total_trades,
        'win_rate': win_rate,
        'avg_return': avg_return,
        'calmar_ratio': calmar_ratio
    }

# ==========================================
# WFO 主执行逻辑
# ==========================================
def run_wfo_optimization(df_raw, wfo_windows):
    print("开始全局计算预热指标...")
    df_prepared = calculate_global_indicators(df_raw)
    
    # 放宽后的参数遍历空间
    param_grid = {
        'theta_buy': np.arange(4.0, 10.0, 1.0),   # 从 4.0 到 9.0
        'theta_sell': np.arange(10.0, 20.0, 2.0),
        'alpha_vol': np.arange(0.6, 1.0, 0.1)     # 放宽缩量比例：0.6 到 0.9
    }
    
    keys, values = zip(*param_grid.items())
    param_combinations = [dict(zip(keys, v)) for v in product(*values)]
    
    wfo_results = {}
    
    for idx, window in enumerate(wfo_windows):
        train_start, train_end = window['train']
        test_start, test_end = window['test']
        print(f"正在处理 WFO 窗口 {idx + 1}: 训练集 {train_start} 至 {train_end}")
        
        # 此时切割出来的训练集已经包含了完整的指标，无冷启动问题
        df_train = df_prepared[(df_prepared['date'] >= train_start) & (df_prepared['date'] <= train_end)]
        
        best_params = None
        best_calmar = -1.0
        best_metrics = {}
        
        # 1. 训练集网格搜索
        for params in param_combinations:
            metrics = evaluate_strategy(df_train, **params)
            # 只有 Calmar > 0（意味着交易次数 >= 10 且有正收益）才会被记录
            if metrics['calmar_ratio'] > best_calmar:
                best_calmar = metrics['calmar_ratio']
                best_params = params
                best_metrics = metrics
        
        # 2. 测试集验证 (Out-of-Sample)
        if best_params is not None:
            print(f"  找到最佳参数: {best_params}, 训练集 Calmar: {best_calmar:.2f}, 交易笔数: {best_metrics['total_trades']}")
            
            df_test = df_prepared[(df_prepared['date'] >= test_start) & (df_prepared['date'] <= test_end)]
            test_metrics = evaluate_strategy(df_test, **best_params)
            print(f"  测试集验证结果: 交易笔数: {test_metrics['total_trades']}, Calmar: {test_metrics['calmar_ratio']:.2f}")
            
            wfo_results[f"Window_{idx+1}"] = {
                'train_period': [train_start, train_end],
                'test_period': [test_start, test_end],
                'best_parameters': best_params,
                'train_metrics': best_metrics,
                'test_metrics': test_metrics
            }
        else:
            print("  警告：该窗口下所有参数的交易次数均不足 10 笔，无法找到有效解！请进一步放宽参数。")

    # ==========================================
    # 调用补丁 1，安全保存为 JSON
    # ==========================================
    safe_results = convert_np_types(wfo_results)
    with open('wfo_tuning_results.json', 'w', encoding='utf-8') as f:
        json.dump(safe_results, f, indent=4, ensure_ascii=False)
        
    print("WFO 优化完成，结果已保存至 wfo_tuning_results.json")

改动亮点清单（便于您去校核代码逻辑）：
calculate_global_indicators 函数：原本您的 Agent 可能是在 `df_traindf_train 上单独 rolling 计算指标。现在我把它提到了最外面，计算完所有技术指标（包括会消耗前面 65 天数据的 `shiftshift(5)）之后，再去通过 date 进行过滤切割。这就像先把面条全煮熟，再分碗，保证测试集里捞出来的第一根面条就是熟的。
强制约束门槛：在 evaluate_strategy 函数中有一行核心拦截代码：if total_trades < 10: calmar_ratio = 0.0。如果 AI 在训练集里搜到了那个一年只交易 1 次的“神级洗盘”参数，它的得分会被无情归零，迫使算法去寻找更普适的常态洗盘参数。
安全导出模块：末尾调用 convert_np_types 进行 JSON 清洗，彻底斩断 int64 报错。
