### 3.2 实现代码

```python
import pandas as pd
import psycopg2

def build_stock_pool():
    """
    构建股票池（80-120只）
    
    Returns:
    --------
    list : 股票代码列表
    """
    conn = psycopg2.connect(**DB_CONFIG)
    
    query = """
        WITH stock_liquidity AS (
            SELECT 
                ts_code,
                AVG(amount) as avg_amount
            FROM kline_daily
            WHERE trade_date >= '20250309'  -- 最近半年
              AND trade_date <= '20260309'
            GROUP BY ts_code
        ),
        stock_market_cap AS (
            SELECT 
                ts_code,
                AVG(total_mv) as avg_market_cap
            FROM daily_basic
            WHERE trade_date >= '20250309'
              AND trade_date <= '20260309'
            GROUP BY ts_code
        )
        SELECT DISTINCT
            sb.ts_code,
            sb.name,
            sb.industry,
            sl.avg_amount,
            smc.avg_market_cap
        FROM stock_basic sb
        INNER JOIN stock_liquidity sl ON sb.ts_code = sl.ts_code
        INNER JOIN stock_market_cap smc ON sb.ts_code = smc.ts_code
        WHERE sb.list_status = 'L'
          AND sb.list_date < '20250309'  -- 上市满1年
          AND (
              sb.industry LIKE '%半导体%' OR
              sb.industry LIKE '%集成电路%' OR
              sb.industry LIKE '%光学光电子%' OR
              sb.industry LIKE '%通信设备%' OR
              sb.industry LIKE '%计算机设备%' OR
              sb.industry LIKE '%自动化设备%' OR
              sb.name LIKE '%机器人%' OR
              sb.name LIKE '%算力%' OR
              sb.name LIKE '%AI%'
          )
          AND sl.avg_amount >= 50000  -- 日均成交额 > 5000万（单位：千元）
          AND smc.avg_market_cap >= 500000  -- 市值 > 50亿（单位：万元）
        ORDER BY smc.avg_market_cap DESC
        LIMIT 120
    """
    
    df = pd.read_sql(query, conn)
    conn.close()
    
    stock_pool = df['ts_code'].tolist()
    
    print(f"✅ 股票池构建完成：{len(stock_pool)} 只股票")
    print(f"   行业分布：{df['industry'].value_counts().to_dict()}")
    
    return stock_pool
```

## 四、模块2：股票过滤

### 4.1 过滤规则

**目标**：从股票池中每日筛选20-40只高质量股票

**规则**：
1. ✅ 财务质量：ROE > 15%、EPS增长 > 20% YoY
2. ✅ 流动性：成交量波动 < 30%、无连续涨停
3. ✅ 趋势：SMA60向上、Close > SMA200 * 0.9
4. ✅ 北向资金：净流入 > 0

### 4.2 实现代码

```python
def filter_stocks(stock_pool, trade_date):
    """
    每日过滤股票（20-40只）
    
    Parameters:
    -----------
    stock_pool : list
        股票池
    trade_date : str
        交易日期（YYYYMMDD）
    
    Returns:
    --------
    list : 过滤后的股票列表
    """
    conn = psycopg2.connect(**DB_CONFIG)
    
    # 转换为SQL IN子句
    stock_codes = "','" .join(stock_pool)
    
    query = f"""
        WITH latest_financial AS (
            SELECT DISTINCT ON (ts_code)
                ts_code,
                roe,
                netprofit_yoy
            FROM fina_indicator
            WHERE ts_code IN ('{stock_codes}')
              AND end_date <= '{trade_date}'
            ORDER BY ts_code, end_date DESC
        ),
        north_flow AS (
            SELECT north_money
            FROM moneyflow_hsgt
            WHERE trade_date = '{trade_date}'
        )
        SELECT 
            k.ts_code,
            k.close,
            t.sma60,
            t.sma200,
            t.vol_sma20,
            k.vol,
            f.roe,
            f.netprofit_yoy
        FROM kline_daily k
        INNER JOIN technical_indicators t 
            ON k.ts_code = t.ts_code AND k.trade_date = t.trade_date
        LEFT JOIN latest_financial f ON k.ts_code = f.ts_code
        CROSS JOIN north_flow nf
        WHERE k.trade_date = '{trade_date}'
          AND k.ts_code IN ('{stock_codes}')
          AND f.roe > 15  -- ROE > 15%
          AND f.netprofit_yoy > 20  -- EPS增长 > 20%
          AND k.close > t.sma200 * 0.9  -- 价格 > SMA200 * 0.9
          AND t.sma60 > LAG(t.sma60, 5) OVER (PARTITION BY k.ts_code ORDER BY k.trade_date)  -- SMA60向上
          AND nf.north_money > 0  -- 北向资金净流入
        ORDER BY f.roe DESC, f.netprofit_yoy DESC
        LIMIT 40
    """
    
    df = pd.read_sql(query, conn)
    conn.close()
    
    filtered_stocks = df['ts_code'].tolist()
    
    print(f"✅ {trade_date}: 过滤后 {len(filtered_stocks)} 只股票")
    
    return filtered_stocks
```

## 五、模块3：量化策略信号生成

### 5.1 买入信号

**规则**（来自原文档）：
1. ✅ SMA60向上（趋势过滤）
2. ✅ BIAS20 < -θ（负乖离率，默认-6%）
3. ✅ Vol < α × Vol_SMA10（缩量，默认0.6倍）
4. ✅ RSI14 < 30（超卖）
5. ✅ MACD HIST转正（HIST > HIST_prev）
6. ✅ 阳线（Close > Open）

### 5.2 卖出信号

**规则**：
1. ✅ BIAS20 > 12%（正乖离率过大）
2. ✅ Close < SMA60 * 0.95（跌破均线5%）
3. ✅ 上影线过长（(High - Close) > 0.05 × (High - Low)）

### 5.3 实现代码

```python
def generate_signals(df, params):
    """
    生成买卖信号
    
    Parameters:
    -----------
    df : pd.DataFrame
        包含K线数据 + 技术指标
    params : dict
        策略参数 {'theta_buy': 6.0, 'alpha_vol': 0.6, 'rsi_thresh': 30, ...}
    
    Returns:
    --------
    pd.DataFrame
        添加 'signal' 列（1=买入, -1=卖出, 0=持有）
    """
    # 买入条件
    buy_cond = (
        (df['sma60'] > df['sma60'].shift(5)) &  # SMA60向上
        (df['bias20'] < -params['theta_buy']) &  # 负乖离率
        (df['vol'] < params['alpha_vol'] * df['vol_sma10']) &  # 缩量
        (df['rsi14'] < params['rsi_thresh']) &  # RSI超卖
        (df['macd_hist'] > df['macd_hist'].shift(1)) &  # MACD HIST转正
        (df['close'] > df['open'])  # 阳线
    )
    
    # 卖出条件
    sell_cond = (
        (df['bias20'] > params['theta_sell']) |  # 正乖离率过大
        (df['close'] < df['sma60'] * 0.95) |  # 跌破均线5%
        ((df['high'] - df['close']) > 0.05 * (df['high'] - df['low']))  # 上影线过长
    )
    
    df['signal'] = 0
    df.loc[buy_cond.shift(1), 'signal'] = 1  # T+1买入
    df.loc[sell_cond.shift(1), 'signal'] = -1  # T+1卖出
    
    return df
```

### 6.2 遗传算法优化

```python
from deap import base, creator, tools
import random

creator.create("FitnessMax", base.Fitness, weights=(1.0,))
creator.create("Individual", list, fitness=creator.FitnessMax)

def evaluate(individual, train_data):
    """
    评估函数：Calmar Ratio - 换手惩罚
    
    Parameters:
    -----------
    individual : list
        [theta_buy, theta_sell, alpha_vol, rsi_thresh]
    train_data : dict
        训练集数据
    
    Returns:
    --------
    tuple : (fitness_score,)
    """
    params = {
        'theta_buy': individual[0],
        'theta_sell': individual[1],
        'alpha_vol': individual[2],
        'rsi_thresh': individual[3]
    }
    
    # 简化回测
    results = simple_backtest(train_data, params)
    
    annual_ret = results['annual_return']
    max_dd = results['max_drawdown']
    turnover = results['turnover']
    
    # Calmar Ratio - 换手惩罚
    calmar = annual_ret / abs(max_dd) if max_dd != 0 else 0
    turnover_penalty = max(0, turnover - 0.15) * 10
    
    fitness = calmar - turnover_penalty
    
    return (fitness,)


def optimize_parameters(train_data):
    """
    遗传算法优化参数
    
    Returns:
    --------
    dict : 最优参数
    """
    toolbox = base.Toolbox()
    
    # 定义个体（4个参数）
    toolbox.register("attr_theta_buy", random.uniform, 4.0, 10.0)
    toolbox.register("attr_theta_sell", random.uniform, 10.0, 20.0)
    toolbox.register("attr_alpha_vol", random.uniform, 0.4, 0.7)
    toolbox.register("attr_rsi_thresh", random.randint, 20, 40)
    
    toolbox.register("individual", tools.initCycle, creator.Individual,
                     (toolbox.attr_theta_buy, toolbox.attr_theta_sell,
                      toolbox.attr_alpha_vol, toolbox.attr_rsi_thresh), n=1)
    
    toolbox.register("population", tools.initRepeat, list, toolbox.individual)
    toolbox.register("evaluate", evaluate, train_data=train_data)
    toolbox.register("mate", tools.cxTwoPoint)
    toolbox.register("mutate", tools.mutGaussian, mu=0, sigma=1, indpb=0.2)
    toolbox.register("select", tools.selTournament, tournsize=3)
    
    # 初始化种群
    pop = toolbox.population(n=50)
    
    # 进化50代
    for gen in range(50):
        offspring = toolbox.select(pop, len(pop))
        offspring = list(map(toolbox.clone, offspring))
        
        # 交叉
        for child1, child2 in zip(offspring[::2], offspring[1::2]):
            if random.random() < 0.5:
                toolbox.mate(child1, child2)
                del child1.fitness.values
                del child2.fitness.values
        
        # 变异
        for mutant in offspring:
            if random.random() < 0.2:
                toolbox.mutate(mutant)
                del mutant.fitness.values
        
        # 评估
        invalid_ind = [ind for ind in offspring if not ind.fitness.valid]
        fitnesses = map(toolbox.evaluate, invalid_ind)
        for ind, fit in zip(invalid_ind, fitnesses):
            ind.fitness.values = fit
        
        pop[:] = offspring
        
        # 打印进度
        fits = [ind.fitness.values[0] for ind in pop]
        print(f"Gen {gen}: Best={max(fits):.4f}, Avg={sum(fits)/len(fits):.4f}")
    
    # 返回最优个体
    best_ind = tools.selBest(pop, 1)[0]
    
    best_params = {
        'theta_buy': best_ind[0],
        'theta_sell': best_ind[1],
        'alpha_vol': best_ind[2],
        'rsi_thresh': int(best_ind[3])
    }
    
    print(f"\n✅ 最优参数：{best_params}")
    
    return best_params
```

## 八、模块6：回测引擎

### 8.1 使用 Backtrader

```python
import backtrader as bt

class QuantStrategy(bt.Strategy):
    params = (
        ('theta_buy', 6.0),
        ('theta_sell', 12.0),
        ('alpha_vol', 0.6),
        ('rsi_thresh', 30),
    )
    
    def __init__(self):
        # 预加载信号（从数据库）
        self.signals = {}  # {ts_code: DataFrame with signals}
    
    def next(self):
        date = self.datetime.date(0)
        
        for data in self.datas:
            ts_code = data._name
            
            if ts_code not in self.signals:
                continue
            
            signal_df = self.signals[ts_code]
            signal_row = signal_df[signal_df['trade_date'] == date.strftime('%Y%m%d')]
            
            if len(signal_row) == 0:
                continue
            
            signal = signal_row['signal'].iloc[0]
            
            if signal == 1:  # 买入
                # 计算仓位（基于ATR）
                atr = signal_row['atr14'].iloc[0]
                risk_capital = self.broker.getvalue() * 0.02  # 单笔风险2%
                shares = int(risk_capital / atr / 100) * 100  # 整百股
                
                self.buy(data=data, size=shares)
                
            elif signal == -1:  # 卖出
                self.close(data=data)


def run_backtest(stock_pool, start_date, end_date, params):
    """
    运行回测
    
    Returns:
    --------
    dict : 回测结果
    """
    cerebro = bt.Cerebro()
    
    # 添加策略
    cerebro.addstrategy(QuantStrategy, **params)
    
    # 加载数据
    for ts_code in stock_pool:
        df = load_stock_data_with_indicators(ts_code, start_date, end_date)
        data = bt.feeds.PandasData(dataname=df, name=ts_code)
        cerebro.adddata(data)
    
    # 设置初始资金
    cerebro.broker.setcash(1000000)
    
    # 设置佣金
    cerebro.broker.setcommission(commission=0.00025, stocklike=True)
    
    # 添加分析器
    cerebro.addanalyzer(bt.analyzers.Returns, _name='returns')
    cerebro.addanalyzer(bt.analyzers.DrawDown, _name='drawdown')
    cerebro.addanalyzer(bt.analyzers.SharpeRatio, _name='sharpe')
    
    # 运行回测
    results = cerebro.run()
    strat = results[0]
    
    # 提取指标
    returns = strat.analyzers.returns.get_analysis()
    drawdown = strat.analyzers.drawdown.get_analysis()
    sharpe = strat.analyzers.sharpe.get_analysis()
    
    backtest_results = {
        'annual_return': returns['rnorm'],
        'max_drawdown': drawdown['max']['drawdown'] / 100,
        'sharpe_ratio': sharpe['sharperatio'],
        'final_value': cerebro.broker.getvalue()
    }
    
    print(f"\n✅ 回测完成：")
    print(f"   年化收益率: {backtest_results['annual_return']:.2%}")
    print(f"   最大回撤: {backtest_results['max_drawdown']:.2%}")
    print(f"   夏普比率: {backtest_results['sharpe_ratio']:.2f}")
    
    # 绘图
    cerebro.plot()
    
    return backtest_results
```
