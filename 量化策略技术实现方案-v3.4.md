w_{\text{bias}} \times \text{clip}\left( \frac{\theta_{\text{buy}} - \text{BIAS20}_t}{8.0}, 0, 1 \right)
$$
$$
w_{\text{vol}} \times \mathbb{I}(\text{vol}t < \alpha{\text{vol}} \times \text{vol_sma10}_t)
w_{\text{rsi}} \times \text{clip}\left( \frac{\text{rsi_thresh} - \text{RSI14}_t}{15.0}, 0, 1 \right)
$$
$$
\text{Buy}_t = \mathbb{I}(\text{score}_t > \text{score_threshold}) \land (\text{pct_change} > -0.095)
$$

from vnpy.app.cta_strategy import CtaTemplate
from vnpy.trader.object import BarData
import pandas as pd
import numpy as np
from quantv3.signal_generator import compute_score_v33  # 导入原有
from quantv3.config import VN_PARAMS

class V33AStockStrategy(CtaTemplate):
    author = "全球量化策略师"
    # 参数（VN.py GUI 可调，与 v3.3 完全一致）
    w_trend: float = 0.25
    w_bias: float = 0.25
    w_vol: float = 0.15
    w_rsi: float = 0.15
    theta_buy: float = -8.0
    theta_sell: float = 10.0
    alpha_vol: float = 0.8
    rsi_thresh: float = 30.0
    score_threshold: float = 0.5

    bars: list = []

    def on_init(self):
        self.write_log("v3.4 VN.py 策略启动 - PostgreSQL 已连接")
        self.load_bar(200)  # 预加载 SMA/RSI 计算

    def on_bar(self, bar: BarData):
        self.bars.append(bar)
        if len(self.bars) < 60: return
        df = pd.DataFrame([{
            'close': b.close_price, 'vol': b.volume,
            'sma60': pd.Series([b.close_price for b in self.bars]).rolling(60).mean().iloc[-1],
            # 其他指标（BIAS20、RSI14、vol_sma10）用 TA-Lib 或您原有函数补全
        } for b in self.bars[-100:]]).tail(1)

        params = {k: getattr(self, k) for k in VN_PARAMS.keys()}
        score, signal, cost = compute_score_v33(df, params)

        if signal.iloc[-1] > 0 and not self.pos:
            self.buy(bar.close_price, 100, slippage=0.0003)  # A股滑点
        elif signal.iloc[-1] < 0 and self.pos > 0:
            self.sell(bar.close_price, self.pos)

        self.write_log(f"Score: {score.iloc[-1]:.4f} | Cost: {cost.iloc[-1]:.4f} | Pos: {self.pos}")
